裴蜀定理

有一对正整数$a,b$，存在非零整数$x,y$，使得 $ax+by = \gcd(a,b)$

使用扩展欧几里得算法构造出$x,y$所以可以证明

int gcd(int a, int b) {

$(a, b) = (b, a \mod b)$这是欧几里得，$(a, 0) = a$ 所以 $ax + 0y = a$ 此时 $x = 1,y = 0$

// Problem: 扩展欧几里得算法
// Contest: AcWing
// URL: https://www.acwing.com/problem/content/879/
// Memory Limit: 64 MB
// Time Limit: 5000 ms
//
// Powered by CP Editor (https://cpeditor.org)

#include <bits/stdc++.h>

using namespace std;

const int inf = 0x3f3f3f3f;

int exgcd(int a, int b, int &x, int &y) {
    if (!b) {
        x = 1, y = 0;
        return a;
    }
    int d = exgcd(b, a % b, y, x);
    y -= a / b * x;
    return d;
}

int main(void) {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    cin >> n;
    while (n--) {
        int a, b;
        cin >> a >> b;
        int x, y;
        exgcd(a, b, x, y);
        cout << x << ' ' << y << endl;
    }

    return 0;
}

// Problem: 表达整数的奇怪方式
// Contest: AcWing
// URL: https://www.acwing.com/problem/content/206/
// Memory Limit: 64 MB
// Time Limit: 1000 ms
//
// Powered by CP Editor (https://cpeditor.org)

#include <bits/stdc++.h>

using namespace std;

const int inf = 0x3f3f3f3f;
typedef long long LL;

LL exgcd(LL a, LL b, LL &x, LL &y) {
    if (!b) {
        x = 1, y = 0;
        return a;
    }
    LL d = exgcd(b, a % b, y, x);
    y -= a / b * x;
    return d;
}

int main(void) {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    cin >> n;
    LL x = 0, m1, a1;
    cin >> a1 >> m1;
    for (int i = 0; i < n - 1; i++) {
        LL m2, a2;
        cin >> a2 >> m2;
        LL k1, k2;
        LL d = exgcd(a1, -a2, k1, k2);
        if ((m2 - m1) % d) {
            cout << "-1" << endl;
            return 0;
        }
        k1 = k1 * (m2 - m1) / d;
        LL t = abs(a2 / d);
        k1 = ((k1 % t) + t) % t;

        m1 = a1 * k1 + m1;
        a1 = abs(a1 / d * a2);
    }

    cout << m1 << endl;

    return 0;
}

#include <bits/stdc++.h>

using namespace std;

const int inf = 0x3f3f3f3f;
const int N = 110;
const double eps = 1e-8;
int n;
double a[N][N];

int gauss() {
    int r, c;
    for (r = 0, c = 0; c < n; c++) {
        int t = r;
        for (int i = r; i < n; i++)
            if (fabs(a[i][c] > fabs(a[t][c]))) t = i;
            
        if (fabs(a[t][c]) < eps) continue;
        
        for (int i = c; i <= n; i++) swap(a[r][i], a[t][i]);
        for (int i = n; i >= c; i--) a[r][i] /= a[r][c];
        for (int i = r + 1; i < n; i++) 
            if (fabs(a[i][c]) > eps) 
                for (int j = n; j >= c; j--) a[i][j] -= a[r][j] * a[i][c];
        
        r++;
    }
    
    if (r < n) {
        for (int i = r; i < n; i++) 
            if (fabs(a[i][n]) > eps) return 2;
        return 1;
    }
    
    for (int i = n - 1; i >= 0; i--)//i是行，j是列
        for (int j = i + 1; j < n; j++) a[i][n] -= a[i][j] * a[j][n];
    return 0;
}

int main(void) {
    // ios::sync_with_stdio(false);
    // cin.tie(0);
    
    cin >> n;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n + 1; j++) {
            cin >> a[i][j];
        }
    }
    
    int t = gauss();
    if (t == 2) cout << "No solution" << endl;
    else if (t == 1) cout << "Infinite group solutions" << endl;
    else {
        for (int i = 0; i < n; i++) {
            if (fabs(a[i][n]) < eps) a[i][n] = 0;
            printf("%.2lf\n", a[i][n]);
        }
    }

    return 0;
}
